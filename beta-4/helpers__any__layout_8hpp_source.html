33 void set_any_layout(const std::vector<dnnl::graph::partition> &partitions,
34  std::unordered_set<size_t> &id_to_set_any_layout) {
35  // mapping from output tensor id to the all supported flags of
36  // supported partitions, we may only need outputs' supported flags
37  std::unordered_map<size_t, std::vector<bool>> output_to_flag_map;
38  for (const auto &p : partitions) {
39  for (const auto &out : p.get_out_ports()) {
40  size_t id = out.get_id();
41  if (p.is_supported()
42  && output_to_flag_map.find(id)
43  == output_to_flag_map.end()) {
44  output_to_flag_map[id] = {};
45  }
46  }
47 
48  for (const auto &in : p.get_in_ports()) {
49  size_t id = in.get_id();
50  auto iter = output_to_flag_map.find(id);
51  if (iter != output_to_flag_map.end()) {
52  // collect all of supported flags of this tensor's uses
53  // Considering we have such a graph:
54  //
55  // partition_A partition_B
56  // \ |
57  // tensor1 tensor2
58  // \ / |
59  // partition_C unsuppported partition
60  // |
61  // tensor3
62  // |
63  // framework op
64  //
65  // so the mapping of partition_A's output will be { true }
66  // the mapping of partition_B's output will be { true, false }
67  // The mapping of partition_C's output will be { false }
68  // Only when all supported flags are true, users can set any
69  // layout.
70  iter->second.push_back(p.is_supported());
71  }
72  }
73  }
74 
75  for (const auto &p : partitions) {
76  // no need to set `any` layout if this partition is not supported
77  if (!p.is_supported()) continue;
78  for (const auto &in : p.get_in_ports()) {
79  size_t id = in.get_id();
80  auto iter = output_to_flag_map.find(id);
81  // if this input tensor is not an output of another supported
82  // partition, just skip
83  if (iter == output_to_flag_map.end()) continue;
84  std::vector<bool> flag_vec = iter->second;
85  // check if all of uses of this tensor are supported partitions,
86  // if not, no need to set ANY layout.
87  bool need_set_any = std::all_of(flag_vec.begin(), flag_vec.end(),
88  [](const bool a) { return a; });
89  if (!need_set_any) continue;
90 
92  id_to_set_any_layout.insert(id);
93  }
94  }
95 }

102 void update_tensors_with_any_layout(
103  std::vector<dnnl::graph::logical_tensor> &lts,
104  const std::unordered_set<size_t> &id_to_set_any_layout) {
105  using dims_t = dnnl::graph::logical_tensor::dims_t;
106  using data_type = dnnl::graph::logical_tensor::data_type;
107  using layout_type = dnnl::graph::logical_tensor::layout_type;
108 
109  for (auto &lt : lts) {
110  size_t id = lt.get_id();
111  if (id_to_set_any_layout.find(id) != id_to_set_any_layout.end()) {
112  dims_t ori_dims = lt.get_dims();
113  data_type ori_dtype = lt.get_data_type();
114  // update old logical tensor with ANY layout
115  lt = dnnl::graph::logical_tensor {
116  id, ori_dtype, ori_dims, layout_type::any};
117  }
118  }
119 }

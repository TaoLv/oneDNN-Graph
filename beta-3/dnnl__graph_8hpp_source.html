115 struct error : public std::exception {
116  dnnl_graph_status_t result;
117  std::string detailed_message;
118 
123  error(dnnl_graph_status_t result, const std::string &message)
124  : result(result)
125  , detailed_message(message + ": " + result2str(result)) {}
126 
131  std::string result2str(dnnl_graph_status_t result) {
132  switch (result) {
133  case dnnl_graph_success: return "success";
134  case dnnl_graph_out_of_memory: return "out of memory";
135  case dnnl_graph_invalid_arguments: return "invalid arguments";
136  case dnnl_graph_unimplemented: return "unimplemented";
137  case dnnl_graph_iterator_ends: return "iterator ends";
138  case dnnl_graph_runtime_error: return "runtime error";
139  case dnnl_graph_not_required: return "not required";
140  case dnnl_graph_invalid_graph: return "invalid graph";
141  case dnnl_graph_invalid_graph_op: return "invalid op";
142  case dnnl_graph_invalid_shape: return "invalid shape";
143  case dnnl_graph_invalid_data_type: return "invalid data type";
144  default: return "unknown error";
145  }
146  }
147 
151  const char *what() const noexcept override {
152  return detailed_message.c_str();
153  }
154 
159  static void check_succeed(
160  dnnl_graph_status_t result, const std::string &message) {
161  if (result != dnnl_graph_success) throw error(result, message);
162  }
163 };

842  error::check_succeed(
843  dnnl_graph_compiled_partition_get_outputs_num(get(), &num),
844  "could not get outputs num from compiled partition");
845  if (num == 0) return {};

848  std::vector<const dnnl_graph_logical_tensor_t *> c_inputs;

851  c_inputs.push_back(&(in.data));

857  context.get()),
858  "could not query dynamic outputs from compiled partition");
859 
860  std::vector<logical_tensor> outputs;
861  outputs.reserve(num);
862  for (auto &c_out : c_outputs)
863  outputs.emplace_back(c_out);
864 
865  return outputs;

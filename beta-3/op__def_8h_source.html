24 static inline example_tensor_t *createConv2d(example_graph_t *graph, char *name,
25  example_tensor_t *input, int32_t out_channels, int64_t kernel_size,
26  int64_t strides, int64_t padding, int64_t dilations, int64_t groups) {
27  int32_t oc = out_channels;
28  int64_t ks[] = {kernel_size, kernel_size};
29  int64_t s[] = {strides, strides};
30  int64_t p[] = {padding, padding};
31  int64_t d[] = {dilations, dilations};
32  int64_t g = groups;
33  example_op_t *op;
34  example_tensor_t *output = conv2d(&op, name, input, oc, ks, s, p, d, &g);
35  CHECK_EXAMPLE(example_graph_add_op(graph, op));
36  return output;
37 }

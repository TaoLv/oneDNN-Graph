48 public:
49  static constexpr auto default_del = del;
50 
52  handle() = default;

54 
60  handle(T t, bool weak = false) { reset(t, weak); }
61 
63  handle(const handle &) = default;
65  handle &operator=(const handle &) = default;
67  handle(handle &&) = default;
69  handle &operator=(handle &&) = default;

77  data_.reset(t, weak ? dummy_del : default_del);

79 
83  T get() const { return data_.get(); }
84

89 };

114 struct error : public std::exception {
115  dnnl_graph_status_t result;
116  std::string detailed_message;
117 
122  error(dnnl_graph_status_t result, const std::string &message)
123  : result(result)
124  , detailed_message(message + ": " + result2str(result)) {}
125 
130  std::string result2str(dnnl_graph_status_t result) {
131  switch (result) {
132  case dnnl_graph_success: return "success";
133  case dnnl_graph_out_of_memory: return "out of memory";
134  case dnnl_graph_invalid_arguments: return "invalid arguments";
135  case dnnl_graph_unimplemented: return "unimplemented";
136  case dnnl_graph_iterator_ends: return "iterator ends";
137  case dnnl_graph_runtime_error: return "runtime error";
138  case dnnl_graph_not_required: return "not required";
139  case dnnl_graph_invalid_graph: return "invalid graph";
140  case dnnl_graph_invalid_graph_op: return "invalid op";
141  case dnnl_graph_invalid_shape: return "invalid shape";
142  case dnnl_graph_invalid_data_type: return "invalid data type";
143  default: return "unknown error";
144  }
145  }
146 
150  const char *what() const noexcept override {
151  return detailed_message.c_str();
152  }
153 
158  static void check_succeed(
159  dnnl_graph_status_t result, const std::string &message) {
160  if (result != dnnl_graph_success) throw error(result, message);
161  }
162 };

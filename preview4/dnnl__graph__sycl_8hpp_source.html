103 inline cl::sycl::event execute(compiled_partition &c_partition, stream &astream,
104  const std::vector<tensor> &inputs, std::vector<tensor> &outputs,
105  const std::vector<cl::sycl::event> &deps = {}) {
106  std::vector<const dnnl_graph_tensor_t *> c_inputs;
107  c_inputs.reserve(inputs.size());
108  for (auto &in : inputs) {
109  c_inputs.push_back(in.get());
110  }
111  std::vector<const dnnl_graph_tensor_t *> c_outputs;
112  c_outputs.reserve(outputs.size());
113  for (auto &out : outputs) {
114  c_outputs.push_back(out.get());
115  }
116 
117  std::vector<void *> c_deps;
118  c_deps.reserve(deps.size());
119  for (auto &event : deps) {
120  c_deps.push_back(const_cast<cl::sycl::event *>(&event));
121  }
122 
123  cl::sycl::event sycl_event;
124  error::check_succeed(
125  dnnl_graph_sycl_interop_compiled_partition_execute(
126  c_partition.get(), astream.get(),
127  static_cast<uint64_t>(c_inputs.size()), c_inputs.data(),
128  static_cast<uint64_t>(c_outputs.size()), c_outputs.data(),
129  static_cast<uint64_t>(c_deps.size()), c_deps.data(),
130  &sycl_event),
131  "could not execute the compiled_partition on a specified sycl "
132  "stream");
133  return sycl_event;
134 }

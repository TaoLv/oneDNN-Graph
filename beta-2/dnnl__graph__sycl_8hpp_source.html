126 inline sycl::event execute(compiled_partition &c_partition, stream &astream,
127  const std::vector<tensor> &inputs, std::vector<tensor> &outputs,
128  const std::vector<sycl::event> &deps = {}) {
129  std::vector<const_dnnl_graph_tensor_t> c_inputs;
130  c_inputs.reserve(inputs.size());
131  for (auto &in : inputs) {
132  c_inputs.push_back(in.get());
133  }
134  std::vector<const_dnnl_graph_tensor_t> c_outputs;
135  c_outputs.reserve(outputs.size());
136  for (auto &out : outputs) {
137  c_outputs.push_back(out.get());
138  }
139 
140  sycl::event sycl_event;
141  error::check_succeed(
142  dnnl_graph_sycl_interop_compiled_partition_execute(
143  c_partition.get(), astream.get(), c_inputs.size(),
144  c_inputs.data(), c_outputs.size(), c_outputs.data(), &deps,
145  &sycl_event),
146  "could not execute the compiled_partition on a specified sycl "
147  "stream");
148  return sycl_event;
149 }

108 // fill the memory according to the given value
109 // src -> target memory buffer
110 // total_size -> total number of bytes of this buffer
111 // val -> fixed value for initialization
112 template <typename T>
113 void fill_buffer(void *src, size_t total_size, int val) {
114  size_t num_elem = static_cast<size_t>(total_size / sizeof(T));
115  T *src_casted = static_cast<T *>(src);
116  // can be implemented through OpenMP
117  for (size_t i = 0; i < num_elem; ++i)
118  *(src_casted + i) = static_cast<T>(val);
119 }

128  const auto &it = knots_.find(arg);

154 template <typename T>
155 static void compare_data(T *dst, T *ref, size_t size, float rtol = 1e-5f,
156  float atol = 1e-6f, bool equal_nan = false) {
157  auto cal_error = [&](const float dst, const float ref) -> bool {
158  const float diff_f32 = dst - ref;
159  const float gap = rtol
160  * (std::abs(ref) > std::abs(dst) ? std::abs(ref)
161  : std::abs(dst))
162  + atol;
163  bool good = std::abs(diff_f32) <= gap;
164  return good;
165  };
166 
167  for (size_t i = 0; i < size; ++i) {
168  if (std::isfinite(dst[i]) && std::isfinite(ref[i])) {
169  const float ref_f32 = static_cast<float>(ref[i]);
170  const float dst_f32 = static_cast<float>(dst[i]);
171  if (!cal_error(dst_f32, ref_f32)) {
172  printf("expected = %s, actual = %s\n",
173  std::to_string(ref[i]).c_str(),
174  std::to_string(dst[i]).c_str());
175  throw std::runtime_error(
176  "output result is not equal to excepted "
177  "results");
178  }
179  } else {
180  bool cond = (dst[i] == ref[i]);
181  if (equal_nan) { cond = std::isnan(dst[i]) && std::isnan(ref[i]); }
182  if (!cond) {
183  printf("expected = %s, actual = %s\n",
184  std::to_string(ref[i]).c_str(),
185  std::to_string(dst[i]).c_str());
186  throw std::runtime_error(
187  "output result is not equal to excepted "
188  "results");
189  }
190  }
191  }
192 }

84  void insert_or_replace(size_t id, const dnnl::graph::tensor &ts) {
85  auto iter = data_.find(id);

113  std::vector<dnnl::graph::tensor> construct_and_initialize_tensors(
114  const std::vector<dnnl::graph::logical_tensor> &lts,
115  dnnl::graph::compiled_partition &c_partition, int value) {
116  std::vector<dnnl::graph::tensor> ret;
117  ret.reserve(lts.size());
118  for (auto &lt : lts) {
119  size_t id = lt.get_id();
120  if (this->has(id)) {
121  // this tensor map has already contained a tensor, just take
122  // it out
123  ret.emplace_back(this->get(id));
124  } else {
125  // this tensor map doesn't contain this tensor, need create from
126  // logical tensor queried from compiled partition
127  dnnl::graph::logical_tensor new_lt
128  = c_partition.query_logical_tensor(id);
129  // allocate and initialize memory buffer
130 #ifdef DNNL_GRAPH_WITH_SYCL
131  void *mem_ptr = cl::sycl::malloc_shared(new_lt.get_mem_size(),
132  q_.get_device(), q_.get_context());
133  buffer_map_[new_lt.get_id()].reset(
134  mem_ptr, deletor {q_.get_context()});
135  EXAMPLE_SWITCH_TYPE(new_lt.get_data_type(), dtype, {
136  fill_buffer<dtype>(q_, mem_ptr,
137  static_cast<size_t>(
138  new_lt.get_mem_size() / sizeof(dtype)),
139  value);
140  });
141 #else
142  void *mem_ptr = malloc(new_lt.get_mem_size());
143  buffer_map_[new_lt.get_id()].reset(mem_ptr, deletor {});
144  EXAMPLE_SWITCH_TYPE(new_lt.get_data_type(), dtype, {
145  fill_buffer<dtype>(mem_ptr, new_lt.get_mem_size(), value);
146  });
147 #endif
148  ret.emplace_back(dnnl::graph::tensor {new_lt, mem_ptr});
149  this->insert_or_replace(new_lt.get_id(), ret.back());
150  }
151  }
152  return ret;
153  }

162  void convert_tensor_with_queried_format(
163  std::vector<dnnl::graph::tensor> &ts,
164  const std::vector<dnnl::graph::logical_tensor> &lts,
165  const dnnl::graph::logical_tensor &queried_lt,
166  dnnl::graph::engine &eng, dnnl::graph::stream &strm) {
167  size_t lid = queried_lt.get_id();
168  auto found_lt_iter = std::find_if(lts.begin(), lts.end(),
169  [lid](const dnnl::graph::logical_tensor &lt) {
170  return lid == lt.get_id();
171  });
172  auto idx = static_cast<size_t>(
173  std::distance(lts.begin(), found_lt_iter));
174 
175  // firstly create a logical tensor with plain format
176  dims_t ori_dims = lts[idx].get_dims();
177  data_type ori_dtype = lts[idx].get_data_type();
178  dnnl::graph::logical_tensor ori_lt {
179  lid, ori_dtype, ori_dims, layout_type::strided};
180 
181  if (!queried_lt.has_same_layout(lts[idx])) {
182 #ifdef DNNL_GRAPH_WITH_SYCL
183  std::unique_ptr<char, deletor> buffer {
184  nullptr, deletor {q_.get_context()}};
185  buffer.reset(static_cast<char *>(
186  cl::sycl::malloc_shared(queried_lt.get_mem_size(),
187  q_.get_device(), q_.get_context())));
188 #else
189  std::unique_ptr<char, deletor> buffer {nullptr, deletor {}};
190  buffer.reset(
191  static_cast<char *>(malloc(queried_lt.get_mem_size())));
192 #endif
193  // create a conversion partition
194  dnnl::graph::op reorder_op {0, dnnl::graph::op::kind::Reorder,
195  {ori_lt}, {queried_lt}, "reorder"};
196  dnnl::graph::partition convert {reorder_op, eng.get_kind()};
197  // compile to compiled partition
198  dnnl::graph::compiled_partition convert_executable
199  = convert.compile({ori_lt}, {queried_lt}, eng);
200  // real tensor with queried layout
201  dnnl::graph::tensor tensor_r {queried_lt, buffer.get()};
202  // execute the conversion
203  convert_executable.execute(strm, {ts[idx]}, {tensor_r});
204  // replace the original tensor with the new one
205  ts[idx] = tensor_r;
206  this->insert_or_replace(lid, tensor_r);
207 #ifdef DNNL_GRAPH_WITH_SYCL
208  buffer_map_[lid].reset(
209  buffer.release(), deletor {q_.get_context()});
210 #else
211  buffer_map_[lid].reset(buffer.release(), deletor {});
212 #endif
213  }
214  }

225  void update_tensor_handle_by_inplace_options(
226  const std::vector<dnnl::graph::tensor> &inputs,
227  std::vector<dnnl::graph::tensor> &outputs,
228  const std::vector<dnnl::graph::logical_tensor> &input_lts,
229  const std::vector<dnnl::graph::logical_tensor> &output_lts,
230  const std::vector<std::pair<size_t, size_t>> &inplace_options) {
231  for (const auto &p : inplace_options) {
232  size_t input_id = p.first;
233  size_t output_id = p.second;
234  auto input_lt_iter
235  = std::find_if(input_lts.begin(), input_lts.end(),
236  [input_id](const dnnl::graph::logical_tensor &lt) {
237  return input_id == lt.get_id();
238  });
239  auto input_lt_idx = static_cast<size_t>(
240  std::distance(input_lts.begin(), input_lt_iter));
241  auto output_lt_iter
242  = std::find_if(output_lts.begin(), output_lts.end(),
243  [output_id](const dnnl::graph::logical_tensor &lt) {
244  return output_id == lt.get_id();
245  });
246  auto output_lt_idx = static_cast<size_t>(
247  std::distance(output_lts.begin(), output_lt_iter));
248 
249  // free original output buffer if it can be in-placed.
250  buffer_map_[output_lts[output_lt_idx].get_id()].reset();
251  void *new_mem_ptr = get_handle_from_tensor(inputs[input_lt_idx],
252  input_lts[input_lt_idx].get_data_type());
253  outputs[output_lt_idx].set_data_handle(new_mem_ptr);
254  inplace_tensor_ids_.insert(output_id);
255  }
256  }
